#PassOneAssembler.java

import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.FileReader;
import java.io.FileWriter;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.Map;

class PassOneAssembler {

    static class MOTEntry {
        String mnemonic;
        String type;  // IS, DL, AD
        int opcode;
        int length;

        MOTEntry(String mnemonic, String type, int opcode, int length) {
            this.mnemonic = mnemonic;
            this.type = type;
            this.opcode = opcode;
            this.length = length;
        }
    }

    static class Symbol {
        String name;
        int address;
        boolean used;

        Symbol(String name, int address) {
            this.name = name;
            this.address = address;
            this.used = false;
        }
    }

    static class Literal {
        String literal;
        int address;

        Literal(String literal) {
            this.literal = literal;
            this.address = -1;
        }
    }

    static Map<String, MOTEntry> MOT = new HashMap<>();
    static LinkedHashMap<String, Symbol> SYMTAB = new LinkedHashMap<>();
    static List<Literal> LITTAB = new ArrayList<>();
    static List<Integer> POOLTAB = new ArrayList<>();
    static List<String> errors = new ArrayList<>();

    static int LC = 0;

    public static void main(String[] args) throws Exception {

        initializeMOT();

        BufferedReader br = new BufferedReader(new FileReader("LPCC/Ass1/New folder/sample.asm"));
        BufferedWriter icWriter = new BufferedWriter(new FileWriter("sample_IC.txt"));

        POOLTAB.add(0);

        String line;

        while ((line = br.readLine()) != null) {

            if (line.trim().isEmpty())
                continue;

            String[] tokens = line.replace(",", "").split("\\s+");

            int index = 0;

            // Label handling
            if (!MOT.containsKey(tokens[0])) {
                String label = tokens[0];

                if (SYMTAB.containsKey(label) && SYMTAB.get(label).address != -1) {
                    errors.add("Duplicate definition of symbol: " + label);
                } else {
                    SYMTAB.put(label, new Symbol(label, LC));
                }
                index++;
            }

            if (index >= tokens.length)
                continue;

            String mnemonic = tokens[index];

            if (!MOT.containsKey(mnemonic)) {
                errors.add("Invalid mnemonic: " + mnemonic);
                continue;
            }

            MOTEntry entry = MOT.get(mnemonic);

            // START
            if (mnemonic.equals("START")) {
                LC = Integer.parseInt(tokens[index + 1]);
                icWriter.write("(AD," + entry.opcode + ") (C," + LC + ")");
                icWriter.newLine();
                continue;
            }

            // LTORG
            if (mnemonic.equals("LTORG")) {
                assignLiterals(icWriter);
                POOLTAB.add(LITTAB.size());
                continue;
            }

            // END
            if (mnemonic.equals("END")) {
                assignLiterals(icWriter);
                icWriter.write("(AD," + entry.opcode + ")");
                icWriter.newLine();
                break;
            }

            // DS
            if (mnemonic.equals("DS")) {
                int size = Integer.parseInt(tokens[index + 1]);
                icWriter.write("(DL," + entry.opcode + ") (C," + size + ")");
                icWriter.newLine();
                LC += size;
                continue;
            }

            // DC
            if (mnemonic.equals("DC")) {
                icWriter.write("(DL," + entry.opcode + ") (C," + tokens[index + 1] + ")");
                icWriter.newLine();
                LC++;
                continue;
            }

            // IS instructions
            icWriter.write("(IS," + entry.opcode + ") ");

            for (int i = index + 1; i < tokens.length; i++) {

                String operand = tokens[i];

                // Literal
                if (operand.startsWith("=")) {
                    int litIndex = addLiteral(operand);
                    icWriter.write("(L," + litIndex + ") ");
                }

                // Symbol
                else if (Character.isLetter(operand.charAt(0))) {

                    if (!SYMTAB.containsKey(operand)) {
                        SYMTAB.put(operand, new Symbol(operand, -1));
                    }

                    SYMTAB.get(operand).used = true;

                    int symIndex = new ArrayList<>(SYMTAB.keySet()).indexOf(operand);
                    icWriter.write("(S," + symIndex + ") ");
                }

                // Constant
                else {
                    icWriter.write("(C," + operand + ") ");
                }
            }

            icWriter.newLine();
            LC += entry.length;
        }

        br.close();
        icWriter.close();

        checkSymbols();

        writeSymbolTable();
        writeLiteralTable();
        writePoolTable();
        writeErrors();

        System.out.println("Pass-1 completed. All output files generated.");
    }

    static void initializeMOT() {

        MOT.put("START", new MOTEntry("START", "AD", 1, 0));
        MOT.put("END", new MOTEntry("END", "AD", 2, 0));
        MOT.put("LTORG", new MOTEntry("LTORG", "AD", 3, 0));

        MOT.put("MOVER", new MOTEntry("MOVER", "IS", 4, 1));
        MOT.put("MOVEM", new MOTEntry("MOVEM", "IS", 5, 1));
        MOT.put("ADD", new MOTEntry("ADD", "IS", 1, 1));
        MOT.put("SUB", new MOTEntry("SUB", "IS", 2, 1));

        MOT.put("DS", new MOTEntry("DS", "DL", 2, 0));
        MOT.put("DC", new MOTEntry("DC", "DL", 1, 0));
    }

    static int addLiteral(String literal) {

        for (int i = 0; i < LITTAB.size(); i++) {
            if (LITTAB.get(i).literal.equals(literal))
                return i;
        }

        LITTAB.add(new Literal(literal));
        return LITTAB.size() - 1;
    }

    static void assignLiterals(BufferedWriter icWriter) throws Exception {

        for (Literal lit : LITTAB) {
            if (lit.address == -1) {
                lit.address = LC;
                icWriter.write("(DL,1) (C," + lit.literal.substring(2, lit.literal.length() - 1) + ")");
                icWriter.newLine();
                LC++;
            }
        }
    }

    static void checkSymbols() {

        for (Symbol sym : SYMTAB.values()) {

            if (sym.address == -1) {
                errors.add("Symbol not declared: " + sym.name);
            }

            if (!sym.used && sym.address != -1) {
                errors.add("Warning: Symbol declared but not used: " + sym.name);
            }
        }
    }

    static void writeSymbolTable() throws Exception {

        BufferedWriter bw = new BufferedWriter(new FileWriter("SYMTAB.txt"));
        bw.write("Index\tSymbol\tAddress\n");

        int index = 0;
        for (Symbol s : SYMTAB.values()) {
            bw.write(index + "\t" + s.name + "\t" + s.address + "\n");
            index++;
        }
        bw.close();
    }

    static void writeLiteralTable() throws Exception {

        BufferedWriter bw = new BufferedWriter(new FileWriter("LITTAB.txt"));
        bw.write("Index\tLiteral\tAddress\n");

        for (int i = 0; i < LITTAB.size(); i++) {
            Literal l = LITTAB.get(i);
            bw.write(i + "\t" + l.literal + "\t" + l.address + "\n");
        }
        bw.close();
    }

    static void writePoolTable() throws Exception {

        BufferedWriter bw = new BufferedWriter(new FileWriter("POOLTAB.txt"));
        bw.write("Pool No\tLiteral Index\n");

        for (int i = 0; i < POOLTAB.size(); i++) {
            bw.write(i + "\t" + POOLTAB.get(i) + "\n");
        }
        bw.close();
    }

    static void writeErrors() throws Exception {

        BufferedWriter bw = new BufferedWriter(new FileWriter("ERRORS.txt"));

        if (errors.isEmpty()) {
            bw.write("No errors found.\n");
        } else {
            for (String err : errors) {
                bw.write(err + "\n");
            }
        }
        bw.close();
    }
}



#PassTwoAssembler.java

import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.FileReader;
import java.io.FileWriter;
import java.util.HashMap;
import java.util.Map;

public class PassTwoAssembler {

    static Map<Integer, Integer> symtab = new HashMap<>();
    static Map<Integer, Integer> littab = new HashMap<>();

    public static void main(String[] args) throws Exception {

        loadSymtab();
        loadLittab();

        BufferedReader icReader = new BufferedReader(new FileReader("D:\\LPCC\\Ass1\\New folder\\sample_IC.txt"));
        BufferedWriter mcWriter = new BufferedWriter(new FileWriter("machine_code.txt"));

        String line;

        while ((line = icReader.readLine()) != null) {

            if (line.contains("AD")) {
                continue;  // Ignore assembler directives
            }

            String machineLine = "";

            String[] parts = line.split("\\)");

            for (String part : parts) {

                part = part.replace("(", "").trim();

                if (part.startsWith("IS")) {
                    machineLine += part.split(",")[1] + " ";
                }

                else if (part.startsWith("S")) {
                    int index = Integer.parseInt(part.split(",")[1]);
                    machineLine += symtab.get(index) + " ";
                }

                else if (part.startsWith("L")) {
                    int index = Integer.parseInt(part.split(",")[1]);
                    machineLine += littab.get(index) + " ";
                }

                else if (part.startsWith("C")) {
                    machineLine += part.split(",")[1] + " ";
                }
            }

            mcWriter.write(machineLine.trim());
            mcWriter.newLine();
        }

        icReader.close();
        mcWriter.close();

        System.out.println("Machine code generated in machine_code.txt");
    }

    static void loadSymtab() throws Exception {

        BufferedReader br = new BufferedReader(new FileReader("D:\\LPCC\\Ass1\\New folder\\SYMTAB.txt"));
        br.readLine(); // skip header

        String line;
        while ((line = br.readLine()) != null) {

            String[] parts = line.split("\\s+");

            int index = Integer.parseInt(parts[0]);
            int address = Integer.parseInt(parts[2]);

            symtab.put(index, address);
        }

        br.close();
    }

    static void loadLittab() throws Exception {

        BufferedReader br = new BufferedReader(new FileReader("D:\\LPCC\\Ass1\\New folder\\LITTAB.txt"));
        br.readLine(); // skip header

        String line;
        while ((line = br.readLine()) != null) {

            String[] parts = line.split("\\s+");

            int index = Integer.parseInt(parts[0]);
            int address = Integer.parseInt(parts[2]);

            littab.put(index, address);
        }

        br.close();
    }
}
