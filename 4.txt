#expr.l

%{
#include "y.tab.h"
#include <stdlib.h>
%}

%%
[0-9]+(\.[0-9]+)?   { yylval.dval = atof(yytext); return NUM; }
[ \t]               ;
\n                  return 0;
.                   return yytext[0];
%%

#expr.y

%{
#include <stdio.h>
#include <stdlib.h>

void yyerror(const char *s);
int yylex();
%}

%union {
    double dval;
}

%token <dval> NUM
%left '+' '-'
%left '*' '/'
%right UMINUS

%type <dval> expr

%%
start : expr          { printf("Result = %g\n", $1); }
      ;

expr  : expr '+' expr { $$ = $1 + $3; }
      | expr '-' expr { $$ = $1 - $3; }
      | expr '*' expr { $$ = $1 * $3; }
      | expr '/' expr { $$ = $1 / $3; }
      | '(' expr ')'  { $$ = $2; }
      | '-' expr %prec UMINUS { $$ = -$2; }
      | NUM           { $$ = $1; }
      ;
%%

int main() {
    printf("Enter arithmetic expression: ");
    yyparse();
    return 0;
}

void yyerror(const char *s) {
    fprintf(stderr, "Error: %s\n", s);
    exit(1);
}


#func.l

%{
#include "y.tab.h"
#include <stdlib.h>
#include <string.h>
%}

%%
[0-9]+(\.[0-9]+)?       { yylval.dval = atof(yytext); return NUM; }
\"[^\"]*\"              { 
                          /* strip quotes */
                          yylval.sval = strdup(yytext + 1);
                          yylval.sval[strlen(yylval.sval)-1] = '\0';
                          return STRING; 
                        }
"sqrt"                  return SQRT;
"strlen"                return STRLEN;
[ \t]                   ;
\n                      return 0;
.                       return yytext[0];
%%


#func.y


%{
#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#include <string.h>

void yyerror(const char *s);
int yylex();
%}

%union {
    double dval;
    char  *sval;
}

%token <dval> NUM
%token <sval> STRING
%token SQRT STRLEN

%type <dval> expr

%%
start : expr          { printf("Result = %g\n", $1); }
      ;

expr  : SQRT '(' expr ')'      { $$ = sqrt($3); }
      | STRLEN '(' STRING ')'  { $$ = (double)strlen($3); free($3); }
      | NUM                    { $$ = $1; }
      ;
%%

int main() {
    printf("Enter expression (e.g. sqrt(25) or strlen(\"India\")): ");
    yyparse();
    return 0;
}

void yyerror(const char *s) {
    fprintf(stderr, "Error: %s\n", s);
    exit(1);
}


#varname.l

%{
#include "y.tab.h"
#include <stdio.h>
#include <stdlib.h>
%}

%%
[a-zA-Z]    return A;
[0-9]       return N;
"_"         return U;
\n          return 0;
.           return yytext[0];
%%


#varname.y

%{
#include <stdio.h>
#include <stdlib.h>

void yyerror(const char *s);
int yylex();
%}

%token A N U

%%
start : var                   { printf("Valid variable name\n"); }
      ;

/* An alphanum chunk: letter, or chunk followed by letter/digit */
chunk : A
      | chunk A
      | chunk N
      ;

/* A var is one chunk, or multiple chunks joined by underscores */
var   : chunk
      | var U chunk
      ;
%%

int main() {
    printf("Enter variable name: ");
    yyparse();
    return 0;
}

void yyerror(const char *s) {
    printf("Invalid variable name\n");
    exit(0);
}






#run.txt
-expr
yacc -d expr.y
lex expr.l
gcc lex.yy.c y.tab.c -o expr -lfl
./expr

-func
yacc -d func.y
lex func.l
gcc lex.yy.c y.tab.c -o func -lfl -lm
./func

-varname
yacc -d varname.y
lex varname.l
gcc lex.yy.c y.tab.c -o varname -lfl
./varname
