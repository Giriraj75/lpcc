/* ============================================================
   Assignment 3b: Lexical Analyzer for Parts of Speech
   (Subset of English Language) — WITH Symbol Table
   Tool: LEX (flex)
   Compile: flex 3b_english_with_symtable.l && gcc lex.yy.c -lfl -o eng_sym
   Run:     ./eng_sym
   ============================================================ */

%{
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

/* ── Symbol Table ── */
#define MAX_SYMBOLS 200
#define MAX_LEN     50

typedef struct {
    char lexeme[MAX_LEN];
    char pos[20];       /* part of speech */
    int  line;
} Symbol;

Symbol symtable[MAX_SYMBOLS];
int    sym_count = 0;
int    line_no   = 1;

/* Insert a new entry (skip duplicates) */
void insert_symbol(const char *lexeme, const char *pos) {
    for (int i = 0; i < sym_count; i++)
        if (strcmp(symtable[i].lexeme, lexeme) == 0) return; /* duplicate */
    if (sym_count >= MAX_SYMBOLS) { fprintf(stderr,"Symbol table full!\n"); return; }
    strncpy(symtable[sym_count].lexeme, lexeme, MAX_LEN - 1);
    strncpy(symtable[sym_count].pos,    pos,    19);
    symtable[sym_count].line = line_no;
    sym_count++;
}

void print_symbol_table() {
    printf("\n╔══════════════════════════════════════════════════════╗\n");
    printf("║              SYMBOL TABLE                            ║\n");
    printf("╠══════╦══════════════════════╦═══════════════╦════════╣\n");
    printf("║  #   ║  Lexeme              ║  Part of Spch ║  Line  ║\n");
    printf("╠══════╬══════════════════════╬═══════════════╬════════╣\n");
    for (int i = 0; i < sym_count; i++)
        printf("║ %4d ║ %-20s ║ %-13s ║ %6d ║\n",
               i+1, symtable[i].lexeme, symtable[i].pos, symtable[i].line);
    printf("╚══════╩══════════════════════╩═══════════════╩════════╝\n");
}

/* ============================================================
   Assignment 3d: Lexical Analyzer for Subset of C Language
   WITH Symbol Table
   Tool: LEX (flex)
   Compile: flex 3d_clang_with_symtable.l && gcc lex.yy.c -lfl -o c_sym
   Run:     ./c_sym < input.c
            OR  ./c_sym   (then type C code, Ctrl+D to end)
   ============================================================ */

%{
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

/* ─────────────────────────────────────────
   Symbol Table Implementation
   ───────────────────────────────────────── */
#define MAX_SYM  300
#define MAX_LEX   64
#define MAX_TYPE  30

typedef struct {
    char lexeme[MAX_LEX];
    char token_type[MAX_TYPE];
    int  line_no;
} SymbolEntry;

SymbolEntry sym_table[MAX_SYM];
int sym_count = 0;
int line_no   = 1;

/* Lookup: returns index if found, -1 otherwise */
int lookup(const char *lex) {
    for (int i = 0; i < sym_count; i++)
        if (strcmp(sym_table[i].lexeme, lex) == 0)
            return i;
    return -1;
}

/* Insert (skip duplicates) */
void insert(const char *lex, const char *type) {
    if (lookup(lex) != -1) return;          /* already present */
    if (sym_count >= MAX_SYM) {
        fprintf(stderr, "ERROR: Symbol table overflow!\n");
        return;
    }
    strncpy(sym_table[sym_count].lexeme,     lex,  MAX_LEX  - 1);
    strncpy(sym_table[sym_count].token_type, type, MAX_TYPE - 1);
    sym_table[sym_count].line_no = line_no;
    sym_count++;
}

void print_symbol_table() {
    printf("\n╔══════════════════════════════════════════════════════════════╗\n");
    printf("║                  SYMBOL TABLE (C Language)                   ║\n");
    printf("╠══════╦═══════════════════════╦══════════════════════╦════════╣\n");
    printf("║  #   ║  Lexeme               ║  Token Type          ║  Line  ║\n");
    printf("╠══════╬═══════════════════════╬══════════════════════╬════════╣\n");
    for (int i = 0; i < sym_count; i++)
        printf("║ %4d ║ %-21s ║ %-20s ║ %6d ║\n",
               i+1, sym_table[i].lexeme,
                    sym_table[i].token_type,
                    sym_table[i].line_no);
    printf("╚══════╩═══════════════════════╩══════════════════════╩════════╝\n");
    printf("  Total entries: %d\n", sym_count);
}
%}
